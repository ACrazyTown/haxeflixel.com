I give you my playstate and my player class. The enemy class is not really different from the player.

Playstate.hx

package; import flixel.FlxG; import flixel.FlxSprite; import flixel.FlxState; import flixel.FlxObject; import flixel.text.FlxText; import flixel.ui.FlxButton; import flixel.ui.FlxBar; import flixel.util.FlxMath; import flixel.util.FlxTimer; import flixel.group.FlxTypedGroup; /** * A FlxState which can be used for the actual gameplay. */ class PlayState extends FlxState { public var _enemy:Enemy; public var _player:Player; public var _entity:FlxTypedGroup<FlxSprite>; public var _healthbarGrp:FlxTypedGroup<FlxBar>; private var _level:TiledLevel; /** * Function that is called up when to state is created to set it up. */ override public function create():Void { // Set a background color FlxG.cameras.bgColor = 0xff131c1b; // Show the mouse (in case it hasn't been disabled) #if !FLX_NO_MOUSE //FlxG.mouse.show(); #end _level = new TiledLevel("assets/data/level.tmx"); add(_level.collideTiles); add(_level.walkTiles); _entity = new FlxTypedGroup<FlxSprite>(); _healthbarGrp = new FlxTypedGroup<FlxBar>(); _player = new Player(16*5, 16*21, this); _enemy = new Enemy(16*20, 16*21, _player, this); _entity.add(cast _player); _entity.add(cast _enemy); add(_entity); add(_healthbarGrp); FlxG.camera.follow(_player); super.create(); } /* * Function that is called when this state is destroyed - you might want to * consider setting all objects this state uses to null to help garbage collection. */ override public function destroy():Void { super.destroy(); } /** * Function that is called once every frame. */ override public function update():Void { super.update(); _entity.sort("y", FlxTypedGroup.ASCENDING); _healthbarGrp.sort("y", FlxTypedGroup.ASCENDING); // Collide with the collidable tile layer if (_level.collideWithLevel(_player)) { _player.moveToNextTile = false; } if (_level.collideWithLevel(_enemy)) { _enemy.moveToNextTile = false; } if (_player.collideWith(_enemy)) { _player.moveToNextTile = false; } FlxG.collide(_player, _enemy); } }

Player.hx

package; import flixel.FlxG; import flixel.FlxObject; import flixel.FlxSprite; import flixel.FlxState; import flixel.ui.FlxBar; /** * ... * @author Bl4ckb0ne */ enum MovePlayer { UP; DOWN; LEFT; RIGHT; } class Player extends FlxSprite { // Movement vars inline static private var MOVEMENT_SPEED:Int = 2; // Static var for the speed inline static private var TILE_SIZE:Int = 16; // Static var for the tiles size (thanks Captain' Obvious) public var moveToNextTile:Bool = false; // Flag used to check if the character is moving public var moveDirection:MovePlayer; // Enum used to hold moving direction // Others vars private var _healthBar:FlxBar; //Obviously the health bar public var isAttacking:Bool = false; // Flag used to check if the character is attacking // Default animations strings public var _idle:String = "idle"; public var _walk:String = "walk"; public var _attack:String = "attack"; public function new(X:Int, Y:Int, state:FlxState) { // X,Y: Starting coordinates super(X, Y); // Smoother movements by enabling subpixel rendering forceComplexRender = true; // Make the player and animations loadGraphic("assets/images/player.png", false, true, 48, 64, true); animation.add("idle", [0], 1, true); animation.add("walk", [1], 1, true); animation.add("attack", [2], 1, true); // Set up the health bar health = 100; _healthBar = new FlxBar(16, 16, FlxBar.FILL_LEFT_TO_RIGHT, 256, 32, this, "health"); _healthBar._scrollFactor.x = _healthBar._scrollFactor.y = 0; state.add(_healthBar); // Change the size of the hitbox setSize(48, 16); offset.set(0, 48); } override public function update():Void { // Check if the player is not moving if (velocity.x == 0) { animation.play(_idle); } super.update(); if (isAttacking == true) { animation.play(_attack); } // Move the player to the next block if (moveToNextTile) { switch(moveDirection) { case UP: y -= MOVEMENT_SPEED; animation.play(_walk); case DOWN: y += MOVEMENT_SPEED; animation.play(_walk); case LEFT: x -= MOVEMENT_SPEED; animation.play(_walk); case RIGHT: x += MOVEMENT_SPEED; animation.play(_walk); } } // Check if the player has now reached the next block if ((x % TILE_SIZE == 0) && (y % TILE_SIZE == 0)) { moveToNextTile = false; } // If down arrow key is pressed. if (FlxG.keys.pressed.DOWN || FlxG.keys.pressed.S && this.isTouching(FlxObject.DOWN) == false) { moveTo(MovePlayer.DOWN); } // Else if up arrow key is pressed. else if (FlxG.keys.pressed.UP || FlxG.keys.pressed.W && this.isTouching(FlxObject.UP) == false ) { moveTo(MovePlayer.UP); } // Else if left arrow key is pressed. else if (FlxG.keys.pressed.LEFT || FlxG.keys.pressed.A) { moveTo(MovePlayer.LEFT); } // Else if right arrow key is pressed. else if (FlxG.keys.pressed.RIGHT || FlxG.keys.pressed.D) { moveTo(MovePlayer.RIGHT); } // Stuff for test } public function moveTo(Direction:MovePlayer):Void { // Check if not already moving. if (!moveToNextTile) { moveDirection = Direction; moveToNextTile = true; } } public function collideWith(obj:FlxObject):Bool { return FlxG.overlap(this, obj); } }
